   \else
   \ifdim\dimexpr#1pt=\z@%
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}

   \else
   \ifdim\dimexpr#1pt>\z@
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}

   \fi\fi\fi\fi
 } \makeatother
