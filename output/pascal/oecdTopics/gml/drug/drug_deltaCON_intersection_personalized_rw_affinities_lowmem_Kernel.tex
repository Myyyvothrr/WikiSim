 \newcommand{\ApplyGradient}[1]{%
  \ifdim\dimexpr#1pt=\@ne pt%

   \else
   \ifdim\dimexpr#1pt=\z@%
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}

   \else
   \ifdim\dimexpr#1pt>\z@
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}
