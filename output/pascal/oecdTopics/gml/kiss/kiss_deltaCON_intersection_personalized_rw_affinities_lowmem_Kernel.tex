   \else
   \ifdim\dimexpr#1pt=\z@%
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}
   \hspace{-0.33em}\colorbox{white}{}
   \else
   \ifdim\dimexpr#1pt>\z@
   \pgfmathsetmacro{\PercentColor}{100.0*(#1-\MinNumber)/(\MaxNumber-\MinNumber)}
   \hspace{-0.33em}\colorbox{white!\PercentColor!SeminarGruen}{}
